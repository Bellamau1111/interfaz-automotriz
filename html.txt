function simulateConnect() {
  logEvent("Conexi√≥n simulada con veh√≠culo", "event-green");
  setInterval(() => {
    const temp = (50 + Math.random() * 10).toFixed(1);
    const volt = (11 + Math.random() * 2).toFixed(1);
    document.getElementById("temp").innerText = `üå°Ô∏è Temperatura: ${temp} ¬∞C`;

function sendCommand(cmd) {
  logEvent(`Comando enviado: ${cmd}`, "event-yellow");
  if (cmd === "LOCK") logEvent("Puertas bloqueadas", "event-red");
  if (cmd === "UNLOCK") logEvent("Puertas desbloqueadas", "event-green");
  if (cmd === "FUEL_OFF") logEvent("Bomba de combustible apagada", "event-red");
  if (cmd === "FUEL_ON") logEvent("Bomba de combustible encendida", "event-green");
}

function logEvent(text, style) {
  const log = document.getElementById("eventLog");
  const time = new Date().toLocaleTimeString().slice(0,5);
  const p = document.createElement("p");
  p.className = style;
  p.innerText = `[${time}] ${text}`;
  log.prepend(p);
  if (log.children.length > 10) log.removeChild(log.lastChild);
}
